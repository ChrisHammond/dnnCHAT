/* if you already had SignalRChat installed, you must run this SQL after installing the dnnCHAT module.

You can run it via the HOST/SQL window as a script.

If you never had SignalRChat installed, you don't need to worry about this.

*/

update {databaseOwner}{objectqualifier}tabmodules
set moduleid = (select moduleid from {databaseOwner}{objectqualifier}modules where moduledefid=(select moduledefid from {databaseOwner}{objectqualifier}moduledefinitions where friendlyname like 'dnnchat%'))
where ModuleId = (select moduleid from {databaseOwner}{objectqualifier}modules where moduledefid=(select moduledefid from {databaseOwner}{objectqualifier}moduledefinitions where friendlyname like 'signalrchat%'))
